# Install

```
npm install -g ezs
```

# Run exemples

```
ezs --plug basics ./csv-to-json.ezs < data/csv
ezs --plug basics ./chunk-counter.ezs < data/csv
ezs --plug basics ./compute.ezs < data/csv
ezs --plug basics ./csv-normalize.ezs < data/csv
ezs --plug basics ./csv-to-json.ezs < data/csv
ezs --plug basics ./json-to-csv.ezs < data/json
ezs --plug basics ./line-counter.ezs < data/csv
ezs --plug basics ./rdf-splitter.ezs < data/atom
ezs --plug basics ./rdf-to-csv.ezs < data/rdf
ezs --plug basics ./skos-to-csv.ezs < data/skos
ezs --plug basics ./xml-to-csv.ezs < data/rss

```
